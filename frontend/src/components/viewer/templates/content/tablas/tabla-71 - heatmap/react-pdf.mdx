import React from "react";
import {
  Document,
  Page,
  StyleSheet,
  Text,
  View
} from "@react-pdf-levelup/core";

const trimestres = ["Q1", "Q2", "Q3", "Q4"];

const heatData: { nombre: string; valores: number[] }[] = [
  { nombre: "Teclado mecánico", valores: [72, 65, 88, 91] },
  { nombre: "Monitor 27\"", valores: [40, 55, 60, 48] },
  { nombre: "Mouse inalámbrico", valores: [85, 90, 95, 99] },
  { nombre: "Docking station", valores: [20, 15, 30, 25] },
  { nombre: "Webcam HD", valores: [58, 62, 70, 66] },
  { nombre: "Silla ergonómica", valores: [45, 50, 68, 72] },
];

function toHex(n: number) {
  const h = Math.max(0, Math.min(255, Math.round(n))).toString(16);
  return h.length === 1 ? "0" + h : h;
}

function heatColor(valor: number) {
  const stops = [
    { v: 0, c: [254, 243, 199] },
    { v: 50, c: [251, 146, 60] },
    { v: 100, c: [185, 28, 28] },
  ];
  let lower = stops[0];
  let upper = stops[stops.length - 1];
  for (let i = 0; i < stops.length - 1; i++) {
    if (valor >= stops[i].v && valor <= stops[i + 1].v) {
      lower = stops[i];
      upper = stops[i + 1];
      break;
    }
  }
  const range = upper.v - lower.v || 1;
  const t = (valor - lower.v) / range;
  const rgb = lower.c.map((c, i) => c + t * (upper.c[i] - c));
  return `#${toHex(rgb[0])}${toHex(rgb[1])}${toHex(rgb[2])}`;
}

function heatTextColor(valor: number) {
  return valor >= 55 ? "#ffffff" : "#3a2b00";
}

const base = StyleSheet.create({
  page: {
    padding: 40,
    fontSize: 10,
    fontFamily: "Helvetica",
    color: "#222222",
  },
  sectionTitle: {
    fontSize: 14,
    fontFamily: "Helvetica-Bold",
    marginTop: 18,
    marginBottom: 6,
  },
  sectionDesc: {
    fontSize: 9.5,
    color: "#666666",
    marginBottom: 10,
    lineHeight: 1.4,
  },
});

const heat = StyleSheet.create({
  table: { display: "flex", flexDirection: "column" },
  headerRow: { display: "flex", flexDirection: "row", marginBottom: 6 },
  labelHeader: { flex: 1.4, fontSize: 9, fontFamily: "Helvetica-Bold", color: "#333333" },
  quarterHeader: { flex: 1, fontSize: 9, fontFamily: "Helvetica-Bold", color: "#333333", textAlign: "center" },
  row: { display: "flex", flexDirection: "row", alignItems: "center", marginBottom: 4 },
  label: { flex: 1.4, fontSize: 9, color: "#333333" },
  cellWrap: { flex: 1, paddingHorizontal: 2 },
  cell: {
    height: 26, display: "flex", alignItems: "center", justifyContent: "center", borderRadius: 3,
  },
  cellText: { fontSize: 8.5, fontFamily: "Helvetica-Bold" },
});

function TablaHeatmap() {
  return (
    <View style={heat.table}>
      <View style={heat.headerRow}>
        <Text style={heat.labelHeader}>Producto</Text>
        {trimestres.map((t) => (
          <Text key={t} style={heat.quarterHeader}>{t}</Text>
        ))}
      </View>
      {heatData.map((h) => (
        <View style={heat.row} key={h.nombre}>
          <Text style={heat.label}>{h.nombre}</Text>
          {h.valores.map((v, i) => (
            <View style={heat.cellWrap} key={i}>
              <View style={[heat.cell, { backgroundColor: heatColor(v) }]}>
                <Text style={[heat.cellText, { color: heatTextColor(v) }]}>{v}%</Text>
              </View>
            </View>
          ))}
        </View>
      ))}
    </View>
  );
}

function TablaHeatmapDoc() {
  return (
    <Document>
      <Page size="A4" style={base.page}>
        <TablaHeatmap />
      </Page>
    </Document>
  );
}

export default TablaHeatmapDoc

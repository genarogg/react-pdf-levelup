import React from "react";
import {

  Document,
  LinearGradient,
  Page,

  Stop,
  StyleSheet,

} from "@react-pdf-levelup/core";

const COLORS = {

  accentPrimary: "#4338ca",

  slate: "#5c6079",
  pageBg: "#ffffff",
  zebra: "#faf9ff",
};

const PAGE_CONTENT_WIDTH = 515;
const TABLE_BORDER_WIDTH = 2;
const TABLE_INNER_WIDTH = PAGE_CONTENT_WIDTH - TABLE_BORDER_WIDTH * 2;
const HEADER_HEIGHT = 26;

interface ProductoFicha {
  campo: string;
  valor: string;
}

const productosGradiente: ProductoFicha[][] = [

    { campo: "Precio", valor: "$310.00" },
  ],
  [
    { campo: "SKU", valor: "SKU-5003" },
    { campo: "Producto", valor: "Lámpara de Piso" },
    { campo: "Categoría", valor: "Iluminación" },
    { campo: "Precio", valor: "$145.00" },

    { campo: "Categoría", valor: "Mobiliario" },
    { campo: "Precio", valor: "$890.00" },
  ],
];

// Anchos de columna en % (deben sumar 100)
const COL = { sku: "20%", producto: "36%", categoria: "24%", precio: "20%" };

const styles = StyleSheet.create({
  page: {
    padding: 40,
    backgroundColor: COLORS.pageBg,
    fontSize: 9.5,
  },
  tableWrapper: {
    width: PAGE_CONTENT_WIDTH,
    borderRadius: 10,
    overflow: "hidden",
    borderWidth: TABLE_BORDER_WIDTH,
    borderColor: COLORS.border,
    borderStyle: "solid",
  },
  headerStack: {
    position: "relative",
    width: TABLE_INNER_WIDTH,
    height: HEADER_HEIGHT,
  },
  headerBackground: {
    position: "absolute",
    top: 0,
    left: 0,
  },
  headerRow: {
    width: TABLE_INNER_WIDTH,
    height: HEADER_HEIGHT,
    flexDirection: "row",
    alignItems: "center",
  },
  headerCell: {
    color: "#fff",
    fontWeight: 700,
    fontSize: 9,
    paddingLeft: 10,
  },
  headerCellRight: {
    color: "#fff",
    fontWeight: 700,
    fontSize: 9,
    paddingRight: 10,
    textAlign: "right",
  },
  bodyRow: {
    width: TABLE_INNER_WIDTH,
    flexDirection: "row",
    alignItems: "center",
    minHeight: 26,
    borderBottomWidth: 1,
    borderBottomColor: COLORS.border,
    borderBottomStyle: "solid",
  },
  bodyRowZebra: {
    backgroundColor: COLORS.zebra,
  },
  cell: {

    color: "#1a1a2e",
  },
  cellSlate: {

  cellRight: {

    fontSize: 9.5,

  },
});

const HeaderGradientBackground: React.FC = () => {

      <Rect
        x="0"
        y="0"
        width={TABLE_INNER_WIDTH}
        height={HEADER_HEIGHT}
        fill="url(#headerGrad)"
      />
    </Svg>
  );
};

const HeaderGradiente: React.FC = () => {
  return (
    <Document
      title="Encabezado en Degradado — react-pdf-levelup"

      language="es-ES"

                {fila[0].valor}
              </Text>
              <Text style={[styles.cell, { width: COL.producto }]}>
                {fila[1].valor}
              </Text>
              <Text style={[styles.cellSlate, { width: COL.categoria }]}>
                {fila[2].valor}
              </Text>
              <Text style={[styles.cellRight, { width: COL.precio }]}>
                {fila[3].valor}
              </Text>
            </View>
          ))}
        </View>
      </Page>
    </Document>
  );
};

export default HeaderGradiente;

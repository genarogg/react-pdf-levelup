```tsx
import React from "react";
import { 
      Col6,
      Container,
      Div,
      H3,
      Layout,
      Row,
      Small
    } from "@react-pdf-levelup/core";
import { 
      QR,
      QRstyle
    } from "@react-pdf-levelup/qr";

const COLORS = {
  ink: '#1a1a2e',
  slate: '#5c6079',
  muted: '#94a0b8',
  border: '#e6e8f0',
  cardBg: '#ffffff',
  accentPrimary: '#4338ca',
};

const card = {
  backgroundColor: COLORS.cardBg,
  border: `1px solid ${COLORS.border}`,
  borderRadius: 10,
  padding: 16,
  height: 230,
  width:230,
  justifyContent: 'center',
};

const frame = { padding: 10, backgroundColor: '#fafafa', borderRadius: 8, marginBottom: 10 };

const eyebrow = (color = COLORS.accentPrimary) => ({
  fontSize: 9,
  color,
  letterSpacing: 2,
});

const Components = ({ data = {} }) => {
  const {
    name = 'Ana García',
    role = 'Directora de Producto',
    brand = 'Estudio Nordika',
    brandUrl = 'https://react-pdf-levelup',
  } = data;

  return (
    <Layout size="A4" backgroundColor="#ffffff">
      <Container>
        <Row>
          {/* Columna izquierda — Tarjeta de Contacto (QR básico) */}
          <Col6>
            <Div style={{ paddingRight: 7, marginBottom: 14 }}>
              <Div style={{ ...card, alignItems: 'center' }}>
                <Small style={eyebrow()}>TARJETA DIGITAL · 1/20</Small>
                <Div style={{...frame}}>
                  <QR  url={brandUrl} size={130} colorDark={COLORS.ink} colorLight="#ffffff" margin={3} />
                </Div>
                <H3 style={{ marginBottom: 2 }}>{name}</H3>
                <Small style={{ color: COLORS.slate }}>{role}</Small>
              </Div>
            </Div>
          </Col6>

          {/* Columna derecha — Identidad Digital (QR estilizado) */}
          <Col6>
            <Div style={{ paddingLeft: 7, marginBottom: 14 }}>
              <Div style={{ ...card, alignItems: 'center' }}>
                <Small style={eyebrow()}>IDENTIDAD DIGITAL · 11/20</Small>
                           <Div style={{...frame}}>
                  <QRstyle
                  
                    url={brandUrl}
                    size={130}
                    dotsOptions={{ color: COLORS.accentPrimary, type: 'rounded' }}
                    backgroundOptions={{ color: '#ffffff' }}
                    cornersSquareOptions={{ type: 'extra-rounded', color: COLORS.accentPrimary }}
                    cornersDotOptions={{ type: 'dot', color: COLORS.accentPrimary }}
                  />
                </Div>
                <H3 style={{ marginBottom: 2 }}>{brand}</H3>
                <Small style={{ color: COLORS.muted }}>Identidad de marca</Small>
              </Div>
            </Div>
          </Col6>
        </Row>
      </Container>
    </Layout>
  );
};

export default Components;
```

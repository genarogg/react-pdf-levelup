# Tabla 1 — Corporativa

```tsx
import React from "react";
import {
  Defs,
  Document,
  LinearGradient,
  Page,
  Rect,
  Stop,
  StyleSheet,
  Svg,
  Text,
  View
} from "@react-pdf-levelup/core";

type Producto = {
  id: string;
  nombre: string;
  categoria: string;
  precio: string;
  estado: "Activo" | "Pendiente" | "Cancelado";
};

const productos: Producto[] = [
  { id: "P-001", nombre: "Teclado mecánico", categoria: "Periféricos", precio: "$49.99", estado: "Activo" },
  { id: "P-002", nombre: "Monitor 27\"", categoria: "Pantallas", precio: "$219.00", estado: "Pendiente" },
  { id: "P-003", nombre: "Mouse inalámbrico", categoria: "Periféricos", precio: "$29.50", estado: "Activo" },
  { id: "P-004", nombre: "Docking station", categoria: "Accesorios", precio: "$89.90", estado: "Cancelado" },
  { id: "P-005", nombre: "Webcam HD", categoria: "Periféricos", precio: "$39.00", estado: "Activo" },
  { id: "P-006", nombre: "Silla ergonómica", categoria: "Mobiliario", precio: "$310.00", estado: "Pendiente" },
];

const glass = StyleSheet.create({
  container: { position: "relative", borderRadius: 12, overflow: "hidden", padding: 4 },
  row: {
    display: "flex", flexDirection: "row", alignItems: "center", justifyContent: "space-between",
    backgroundColor: "rgba(255,255,255,0.62)", borderWidth: 1, borderColor: "#ffffff",
    borderRadius: 10, padding: 10, marginBottom: 8,
  },
  nombre: { fontSize: 9.5, fontFamily: "Helvetica-Bold", color: "#1e1b4b" },
  categoria: { fontSize: 8, color: "#4c1d95" },
  precio: { fontSize: 10, fontFamily: "Helvetica-Bold", color: "#1e1b4b" },
});

function GlassBackdrop() {
  return (
    <Svg
      viewBox="0 0 500 340"
      width="100%"
      height="100%"
      preserveAspectRatio="none"
      style={{ position: "absolute", top: 0, left: 0 }}
    >
      <Defs>
        <LinearGradient id="glassGrad" x1="0" y1="0" x2="1" y2="1">
          <Stop offset="0" stopColor="#c7d2fe" stopOpacity={1} />
          <Stop offset="0.5" stopColor="#fbcfe8" stopOpacity={1} />
          <Stop offset="1" stopColor="#a7f3d0" stopOpacity={1} />
        </LinearGradient>
      </Defs>
      <Rect x={0} y={0} width={500} height={340} fill="url(#glassGrad)" />
    </Svg>
  );
}

function TablaGlassmorphism() {
  return (
    <View style={glass.container}>
      <GlassBackdrop />
      {productos.map((p) => (
        <View style={glass.row} key={p.id}>
          <View>
            <Text style={glass.nombre}>{p.nombre}</Text>
            <Text style={glass.categoria}>{p.categoria}</Text>
          </View>
          <Text style={glass.precio}>{p.precio}</Text>
        </View>
      ))}
    </View>
  );
}

const base = StyleSheet.create({
  page: { padding: 40, fontSize: 10, fontFamily: "Helvetica", color: "#222222" },
});

function GlassmorphismDoc() {
  return (
    <Document>
      <Page size="A4" style={base.page}>
        <TablaGlassmorphism />
      </Page>
    </Document>
  );
}

export default GlassmorphismDoc
```
